Object Programming in JavaScript
~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
// Object Initializer
var  obj =  {sid :  1025, sname  : "Scott"};

// Using constructor function 
function   Student()
{
		this.sid  = 1025;
		this.sname  = "Scott";
}


var  obj =  new Student();
